95/**
96 * @brief SysTick Timer Interrupt Service Routine (100 ms period)
97 * @details Core real-time data acquisition routine:
98 * 1. Increments `ticks` counter
99 * 2. Queries MAX30101 FIFO for new samples
100 * 3. If samples available: reads and converts to nanoamps in one call
101 * 4. Toggles status LED (visual feedback)
102 *
103 * This ISR runs non-preemptively (highest priority) approximately every
104 * 100 milliseconds, making it ideal for time-critical sensor polling.
105 *
106 * @param None
107 * @return void
108 * @note ISR Context
109 * - Execution time: ~1–2 ms (I2C reads dominate; ~0.5 ms per sample)
110 * - Called at SysTick exception (cannot nest itself)
111 * - All registers preserved; no clobbering of main loop state
112 *
113 * @data_output
114 * Upon samples available:
115 * - Populates MAX30101_SampleCurrentBuffer[] with up to 32 calibrated samples
116 * - Sample count in local variable `available_samples`
117 * - Data remains in buffer until next ISR overwrites (100 ms window)
118 *
119 * @timing
120 * - Sample freshness: 0–100 ms (age of data in buffer)
121 * - FIFO latency: Variable; depends on sample rate (100 Hz) and read interval
122 * - At 100 Hz rate with 100 ms polling: Expect 8–10 samples per interrupt
123 *
124 * @warning
125 * - Race condition possible if main loop reads buffer while ISR writes
126 * (Consider using volatile or double-buffering in production)
127 * - I2C blocking: If I2C bus is busy, ISR may delay by several ms
128 *
129 * @see MAX30101_GetNumAvailableSamples, MAX30101_ReadFIFO_Current, LED_Toggle
130 * @example
131 * // ISR fires every 100 ms:
132 * // ticks++ (incremented 10 times per second)
133 * // MAX30101_SampleCurrentBuffer[] updated with fresh nA values
134 * // LED toggles (100 ms on, 100 ms off = 5 Hz blink)
135 */
136
137void SysTick_Handler(void) {
138 ticks++;
139 uint8_t available_samples = MAX30101_GetNumAvailableSamples();
140 if (available_samples > 0) {
141 // Read available samples from the MAX30101 FIFO in float format (nA) into the global buffer
142 MAX30101_ReadFIFO_Current(MAX30101_SampleCurrentBuffer, available_samples);
143 }
144 LED_Toggle();
145
146}

148/**
149 * @brief Configure STM32F303K8 system clock to 64 MHz via PLL
150 * @details PLL configuration chain:
151 * - **Input**: 8 MHz HSI oscillator (internal, always available)
152 * - **Divider**: /2 in PLL block (built-in)
153 * - **Multiplier**: MUL[3:0] = 0x0E (multiply by 16)
154 * - **Output**: (8 MHz / 2) × 16 = 64 MHz
155 * - **System Clock**: PLL output becomes SYSCLK
156 * - **Flash Timing**: 2 wait states for 48 < HCLK ≤ 72 MHz
157 * - **APB1 Divider**: HCLK/2 (32 MHz for most peripherals, incl. I2C)
158 *
159 * @param None
160 * @return void
161 *
162 * @operations
163 * 1. RCC->CFGR |= 0xE<<18 (PLLMUL configuration)
164 * 2. FLASH->ACR |= 0x2 (Latency = 2 cycles)
165 * 3. RCC->CR |= RCC_CR_PLLON (Enable PLL oscillator)
166 * 4. Wait for RCC_CR_PLLRDY flag
167 * 5. RCC->CFGR |= 0x402 (SW[1:0]=10 for PLL, PPRE1[2:0]=100 for APB1/2)
168 * 6. Wait for RCC_CFGR_SWS_PLL status
169 * 7. SystemCoreClockUpdate() (Update HAL clock variable)
170 *
171 * @timing
172 * - PLL lock time: ~100 µs (measured in hardware)
173 * - Total configuration time: <1 ms
174 * - Blocking: Yes (waits for PLL_RDY and SWS flags)
175 *
176 * @side_effects
177 * - SYSCLK becomes 64 MHz (all core and bus clocks scale accordingly)
178 * - I2C1 clock: 32 MHz / (I2C prescaler) ≈ 400 kHz
179 * - SysTick frequency scales to HCLK (timer reload values adapt)
180 * - Power consumption increases (~60 mA typical at 64 MHz vs 30 mA at 8 MHz)
181 *
182 * @warning
183 * - PLL must be configured BEFORE any I2C or timer operations
184 * - Changing clock mid-operation may corrupt ongoing communications
185 * - Requires FLASH latency adjustment or reads may fail
186 *
187 * @perf_note
188 * - 64 MHz is ~8× faster than default 8 MHz HSI
189 * - I2C bus speed 400 kHz achievable only with sufficient SYSCLK
190 * - Sampling rate limited more by I2C than by CPU
191 *
192 * @example
193 * // Called from main() to enable high-speed operation
194 * clk_config();
195 * // Now SYSCLK = 64 MHz, I2C/SPI significantly faster
196 */
197void clk_config(void){
198 // PLLMUL <- 0x0E (PLL input clock x 16 --> (8 MHz / 2) * 16 = 64 MHz )
199 RCC->CFGR |= 0xE<<18;
200 // Flash Latency, two wait states for 48<HCLK<=72 MHz
201 FLASH->ACR |= 0x2;
202 // PLLON <- 0x1
203 RCC->CR |= RCC_CR_PLLON;
204 while (!(RCC->CR & RCC_CR_PLLRDY));
205 // SW<-0x02 (PLL as System Clock), HCLK not divided, PPRE1<-0x4 (APB1 <- HCLK/2), APB2 not divided
206 RCC->CFGR |= 0x402;
207 while (!(RCC->CFGR & RCC_CFGR_SWS_PLL));
208 SystemCoreClockUpdate();
209}
