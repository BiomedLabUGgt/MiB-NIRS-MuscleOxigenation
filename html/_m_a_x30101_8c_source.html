6 * @brief Initialize MAX30101 in SpO2 mode (dual-LED configuration)
7 * @details Configures sensor for blood oxygen (SpO2) measurement with low power consumption.
8 * - Mode: SpO2 (Red + IR LEDs)
9 * - Sample Rate: 50 Hz
10 * - ADC Resolution: 16-bit
11 * - FIFO Configuration: Averaging 8, rollover enabled
12 * - LED Power: Low (minimal power draw, suitable for wearables)
13 * - Temperature Sensor: Enabled
14 * @param None
15 * @return void
16 * @note Suitable for battery-powered wearable applications.
17 * Call this once during initialization before reading samples.
18 * @see MAX30101_InitMuscleOx
19 * @example
20 * MAX30101_InitSPO2Lite();
21 * uint8_t samples = MAX30101_GetNumAvailableSamples();
22 */
23void MAX30101_InitSPO2Lite(void){
24 I2C1_Write(SENSOR_ADDR, FIFO_CONFIG, 0x4F); // FIFO avg 8, FIFO rollover enabled
25 I2C1_Write(SENSOR_ADDR, MODE_CONFIG, 0x03); // SPO2 Mode (Red + IR)
26 I2C1_Write(SENSOR_ADDR, SPO2_CONFIG, 0x23); // 2048 range, sample rate 50 Hz, pulse width 215 (16 bits)
27 I2C1_Write(SENSOR_ADDR, FIFO_READPTR, 0x0); // Reset FIFO read pointer
28 I2C1_Write(SENSOR_ADDR, FIFO_WRITPTR, 0x0); // Reset FIFO write pointer
29 I2C1_Write(SENSOR_ADDR, LED1_PAMPLI, 0x18); // Red LED power (low)
30 I2C1_Write(SENSOR_ADDR, LED2_PAMPLI, 0x18); // IR LED power (low)
31 I2C1_Write(SENSOR_ADDR, DIE_TEMPCFG, 0x01); // Enable temperature sensor
32}

34/**
35 * @brief Initialize MAX30101 for NIRS muscle oxygenation measurement
36 * @details Configures sensor for Near-Infrared Spectroscopy with 3 simultaneous LEDs.
37 * Optimal for non-invasive tissue oxygenation and perfusion monitoring.
38 * - Mode: Multi-LED (Red + IR + Green)
39 * - Sample Rate: 100 Hz
40 * - ADC Resolution: 16-bit
41 * - FIFO Configuration: Averaging 8, rollover enabled
42 * - Temperature Sensor: Enabled
43 * @param ledPower - LED current control register value (0x00 to 0xFF)
44 * Range: 4.4 mA to 50.6 mA in ~0.2 mA steps
45 * Typical: 0x4B (~20 mA), 0x18 (~10 mA) for low power
46 * @return void
47 * @note Higher sample rate (100 Hz) vs SPO2Lite (50 Hz) provides better temporal resolution.
48 * Three-LED configuration increases tissue penetration depth for muscle assessment.
49 * @see MAX30101_InitSPO2Lite
50 * @example
51 * MAX30101_InitMuscleOx(0x4B); // 20 mA LED power
52 * uint8_t available = MAX30101_GetNumAvailableSamples();
53 * MAX30101_ReadFIFO_Current(current_buffer, available);
54 */
55void MAX30101_InitMuscleOx(uint8_t ledPower){
56 I2C1_Write(SENSOR_ADDR, FIFO_CONFIG, 0x4F); // FIFO avg 8, FIFO rollover enabled
57 I2C1_Write(SENSOR_ADDR, MODE_CONFIG, 0x07); // Multi-LED mode (Red + IR + Green)
58 I2C1_Write(SENSOR_ADDR, SPO2_CONFIG, 0x26); // 2048 range, sample rate 100 Hz, pulse width 215 (16 bits)
59 I2C1_Write(SENSOR_ADDR, FIFO_READPTR, 0x0); // Reset FIFO read pointer
60 I2C1_Write(SENSOR_ADDR, FIFO_WRITPTR, 0x0); // Reset FIFO write pointer
61 I2C1_Write(SENSOR_ADDR, LED1_PAMPLI, ledPower); // Red LED power (medium)
62 I2C1_Write(SENSOR_ADDR, LED2_PAMPLI, ledPower); // IR LED power (medium)
63 I2C1_Write(SENSOR_ADDR, LED3_PAMPLI, ledPower); // Green LED power (medium)
64 I2C1_Write(SENSOR_ADDR, DIE_TEMPCFG, 0x01); // Enable temperature sensor
65}

67/**
68 * @brief Query FIFO status from MAX30101 sensor
69 * @details Reads FIFO write and read pointer registers to determine number of unread samples.
70 * Accounts for circular 32-sample FIFO with pointer wrap-around.
71 * @param None
72 * @return uint8_t Number of complete samples available (0 to 32)
73 * - Returns 0 if FIFO empty or pointers equal
74 * - Returns 1 to 32 for available samples
75 * @note Call this before MAX30101_ReadFIFO() or MAX30101_ReadFIFO_Current()
76 * to check if new data is ready.
77 * @warning Multiple fast consecutive reads may show inconsistent results
78 * due to FIFO updates during pointer reads.
79 * @see MAX30101_ReadFIFO, MAX30101_ReadFIFO_Current
80 * @example
81 * uint8_t count = MAX30101_GetNumAvailableSamples();
82 * if (count > 0) {
83 * MAX30101_ReadFIFO_Current(samples, count);
84 * }
85 */
86uint8_t MAX30101_GetNumAvailableSamples(void){
87 uint8_t write_ptr = 0;
88 uint8_t read_ptr = 0;
89 uint8_t num_samples = 0;
90
91 // Read FIFO write pointer and read pointer from sensor
92 I2C1_Read(SENSOR_ADDR, FIFO_WRITPTR, &write_ptr, 1);
93 I2C1_Read(SENSOR_ADDR, FIFO_READPTR, &read_ptr, 1);
94
95 // Mask to 5 bits since FIFO pointers are 5 bits (0-31)
96 write_ptr &= 0x1F;
97 read_ptr &= 0x1F;
98
99 // Calculate available samples
100 if (write_ptr >= read_ptr) {
101 num_samples = write_ptr - read_ptr;
102 } else {
103 num_samples = (32 - read_ptr) + write_ptr;
104 }
105
106 return num_samples;
107}

109/**
110 * @brief Read raw FIFO data from MAX30101 (intermediate format)
111 * @details Streams raw 2-byte ADC samples sequentially from each channel.
112 * Data is organized as: Red(2B) IR(2B) Green(2B) repeating.
113 * Each complete sample requires 6 bytes from FIFO.
114 * @param samples - [out] Pointer to buffer of MAX30101_Sample structures to populate
115 * @param num_samples - [in] Number of complete samples to read (max 32)
116 * @return void
117 * @retval N/A
118 * @note This is the lowest-level FIFO read. For processed data in nanoamps,
119 * use MAX30101_ReadFIFO_Current() instead to skip intermediate conversion.
120 * @warning Ensure buffer has capacity for num_samples structures.
121 * @see MAX30101_ReadFIFO_Current, MAX30101_ConvertSampleToUint16
122 * @example
123 * MAX30101_Sample raw[32];
124 * MAX30101_ReadFIFO(raw, 5); // Read 5 samples
125 */
126void MAX30101_ReadFIFO(MAX30101_Sample *samples, uint8_t num_samples){
127 uint8_t i = 0;
128 uint8_t fifo_data[8]; // 6 bytes per complete sample (3 channels × 2 bytes)
129
130 for (i = 0; i < num_samples; i++) {
131 // Read 6 consecutive bytes from FIFO_DATAREG
132 // Each I2C read from address 0x07 gets the next byte in the FIFO
133 I2C1_Read(SENSOR_ADDR, FIFO_DATAREG, fifo_data, 6);
134
135 // Store the 2-byte samples for each channel
136 samples[i].red[0] = fifo_data[0];
137 samples[i].red[1] = fifo_data[1];
138
139 samples[i].ir[0] = fifo_data[2];
140 samples[i].ir[1] = fifo_data[3];
141
142 samples[i].green[0] = fifo_data[4];
143 samples[i].green[1] = fifo_data[5];
144 }
145}

147/**
148 * @brief Convert raw byte pairs to 16-bit ADC counts
149 * @details Combines 2 raw bytes (MSB, LSB) into 16-bit unsigned integers per channel.
150 * No scaling applied—output is raw ADC count (0–65535).
151 * @param sample_in - [in] Pointer to MAX30101_Sample with raw 2-byte data
152 * @param sample_out - [out] Pointer to MAX30101_SampleData for converted counts
153 * @return void
154 * @retval N/A
155 * @note This is typically an intermediate step. For direct nanoamp values,
156 * use MAX30101_ConvertUint16ToCurrent() after this or use
157 * MAX30101_ReadFIFO_Current() to bypass intermediate storage.
158 * @see MAX30101_ConvertUint16ToCurrent, MAX30101_ReadFIFO_Current
159 * @example
160 * MAX30101_Sample raw;
161 * MAX30101_SampleData counts;
162 * MAX30101_ReadFIFO(&raw, 1);
163 * MAX30101_ConvertSampleToUint16(&raw, &counts); // counts.red = 0-65535
164 */
165void MAX30101_ConvertSampleToUint16(MAX30101_Sample *sample_in, MAX30101_SampleData *sample_out){
166 // Convert Red LED: combine 2 bytes into 16-bit uint16_t
167 sample_out->red = ((uint16_t)sample_in->red[0] << 8) |
168 ((uint16_t)sample_in->red[1]);
169
170 // Convert IR LED: combine 2 bytes into 16-bit uint16_t
171 sample_out->ir = ((uint16_t)sample_in->ir[0] << 8) |
172 ((uint16_t)sample_in->ir[1]);
173
174 // Convert Green LED: combine 2 bytes into 16-bit uint16_t
175 sample_out->green = ((uint16_t)sample_in->green[0] << 8) |
176 ((uint16_t)sample_in->green[1]);
177}

179/**
180 * @brief Scale ADC counts to calibrated current (nanoamps)
181 * @details Converts 16-bit ADC values to current using:
182 * Current (nA) = ADC_Count × 0.00781 nA
183 * Full range: 0 to 2048 nA
184 * LSB Resolution: 7.81 pA per count
185 * @param sample_in - [in] Pointer to MAX30101_SampleData with ADC counts (0–65535)
186 * @param sample_out - [out] Pointer to MAX30101_SampleCurrent for current in nanoamps
187 * @return void
188 * @retval N/A
189 * @note Intermediate step for two-phase conversion pipeline. For single-step
190 * FIFO read + conversion, use MAX30101_ReadFIFO_Current() instead.
191 * @see MAX30101_ReadFIFO_Current
192 * @example
193 * MAX30101_SampleData adc_counts;
194 * MAX30101_SampleCurrent current;
195 * MAX30101_ConvertUint16ToCurrent(&adc_counts, &current);
196 * printf("Red LED: %f nA\n", current.red); // Output in nanoamps
197 */
198void MAX30101_ConvertUint16ToCurrent(MAX30101_SampleData *sample_in, MAX30101_SampleCurrent *sample_out){
199 // Convert Red channel ADC count to current in nanoamps
200 sample_out->red = (float32_t)sample_in->red * MAX30101_CURRENT_LSB_NA;
201
202 // Convert IR channel ADC count to current in nanoamps
203 sample_out->ir = (float32_t)sample_in->ir * MAX30101_CURRENT_LSB_NA;

209/**
210 * @brief Optimized single-step FIFO read with direct nanoamp conversion
211 * @details Combines all processing in one loop:
212 * 1. Read 6 raw bytes from FIFO per sample
213 * 2. Combine bytes to 16-bit ADC counts
214 * 3. Scale to current (nA) using LSB=7.81 pA
215 * Output directly in calibrated units without intermediate storage.
216 * @param samples - [out] Pointer to buffer of MAX30101_SampleCurrent to populate
217 * @param num_samples - [in] Number of complete samples to read and convert (max 32)
218 * @return void
219 * @retval N/A
220 * @note RECOMMENDED function for typical usage. Minimizes memory usage and processing
221 * overhead compared to three-step pipeline (Read → uint16 → nanoamps).
222 * @warning Ensure output buffer has capacity for num_samples structures.
223 * Total bytes read from FIFO: num_samples × 6
224 * @performance ~15-20% faster and ~24 bytes less stack vs separate conversions
225 * @see MAX30101_GetNumAvailableSamples
226 * @example
227 * MAX30101_SampleCurrent buf[32];
228 * uint8_t available = MAX30101_GetNumAvailableSamples();
229 * if (available) {
230 * MAX30101_ReadFIFO_Current(buf, available);
231 * // buf[0].red contains Red LED current in nanoamps
232 * // buf[0].ir contains IR LED current in nanoamps
233 * // buf[0].green contains Green LED current in nanoamps
234 * }
235 */
236void MAX30101_ReadFIFO_Current(MAX30101_SampleCurrent *samples, uint8_t num_samples){
237 uint8_t fifo_data[6];
238 uint8_t i;
239 uint16_t temp;
240
241 for(i = 0; i < num_samples; i++){
242 // read 6 bytes from FIFO
243 I2C1_Read(SENSOR_ADDR, FIFO_DATAREG, fifo_data, 6);
244 // combine bytes and convert right away
245 temp = ((uint16_t)fifo_data[0] << 8) | fifo_data[1];
246 samples[i].red = (float32_t)temp * MAX30101_CURRENT_LSB_NA;
247
248 temp = ((uint16_t)fifo_data[2] << 8) | fifo_data[3];
249 samples[i].ir = (float32_t)temp * MAX30101_CURRENT_LSB_NA;
250
251 temp = ((uint16_t)fifo_data[4] << 8) | fifo_data[5];
252 samples[i].green = (float32_t)temp * MAX30101_CURRENT_LSB_NA;
253 }
