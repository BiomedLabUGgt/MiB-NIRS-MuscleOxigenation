5/**
6 * @brief Initialize I2C1 peripheral and GPIO pins for 400 kHz master-mode operation
7 * @details Complete I2C1 setup sequence:
8 * 1. Clock enables (I2C1, GPIOB)
9 * 2. GPIO configuration (open-drain, alternate function AF4)
10 * 3. I2C peripheral reset
11 * 4. Timing register configuration for 400 kHz
12 * 5. Enable I2C1
13 *
14 * ### GPIO Configuration (PB6 = SCL, PB7 = SDA)
15 * - MODER: [14:12]=10 (Alternate Function), [14:12]=10
16 * - OTYPER: [7]=1 (Open-drain), [6]=1
17 * - AFR[0]: [31:28]=0100 (AF4), [27:24]=0100 (AF4)
18 *
19 * ### I2C TIMINGR = 0x00C50F26 (for APB1 = 32 MHz, target 400 kHz)
20 * - PRESC[3:0] = 0 (Prescaler = 1)
21 * - SCLDEL[3:0] = 1010 (Setup delay ≈ 640 ns)
22 * - SDADEL[3:0] = 0101 (Hold delay ≈ 320 ns)
23 * - SCLH[7:0] = 00001111 (SCL high ≈ 1.5 µs)
24 * - SCLL[7:0] = 00100110 (SCL low ≈ 1.0 µs)
25 * - Calculated frequency: ~400 kHz (I2C Fast-mode)
26 *
27 * @param None
28 * @return void
29 * @retval N/A
30 *
31 * @timing
32 * - Total init time: <1 ms
33 * - I2C1 ready after: function completes
34 * - First I2C transaction: can follow immediately
35 *
36 * @critical_sequence
37 * - I2C clock MUST be enabled before TIMINGR modification
38 * - GPIO pins must be configured as open-drain (not push-pull)
39 * - Reset sequence (RSTR flag) clears any prior error states
40 * - TIMINGR is APB1-frequency-sensitive; changing APB1 requires recalculation
41 *
42 * @side_effects
43 * - PB6 and PB7 become I2C1 pins (unavailable for GPIO)
44 * - I2C1 clock enabled in RCC->APB1ENR
45 * - I2C1 will hold SCL/SDA low if bus is busy
46 *
47 * @warning
48 * - Call once at system startup, BEFORE any I2C_Write/Read operations
49 * - Do not change TIMINGR if APB1 frequency changes dynamically
50 * - External pull-up resistors (~4.7 kΩ) required on SCL/SDA lines
51 *
52 * @see I2C1_Write, I2C1_Read
53 */
54
55void I2C1_Config(void) {
56 // Enable I2C1 clock
57 RCC->APB1ENR |= RCC_APB1ENR_I2C1EN;
58 // Enable GPIOB clock
59 RCC->AHBENR |= RCC_AHBENR_GPIOBEN;
60 // Configure PB6 (SCL) and PB7 (SDA) as open-drain alternate function
61 // Clear mode bits for PB6 and PB7
62 GPIOB->MODER &= ~((3 << 12) | (3 << 14));
63 // Set to alternate function (10)
64 GPIOB->MODER |= (2 << 12) | (2 << 14);
65 // Set open-drain output type
66 GPIOB->OTYPER |= (1 << 6) | (1 << 7);
67 // Set alternate function to I2C1 (AF4 for PB6 and PB7)
68 GPIOB->AFR[0] &= ~((0xF << 24) | (0xF << 28));
69 GPIOB->AFR[0] |= (4 << 24) | (4 << 28);
70 // Reset I2C1
71 RCC->APB1RSTR |= RCC_APB1RSTR_I2C1RST;
72 RCC->APB1RSTR &= ~RCC_APB1RSTR_I2C1RST;
73 // Disable I2C1 to configure it
74 I2C1->CR1 &= ~I2C_CR1_PE;
75 // TIMINGR register for 400 kHz with APB1 = 32 MHz
76 // (PRESC=0, SCLDEL=12, SDADEL=5, SCLH=15, SCLL=38)
77 I2C1->TIMINGR = 0x00C50F26;
78 // Enable I2C1
79 I2C1->CR1 |= I2C_CR1_PE;
80}

83/**
84 * @brief Master write single 8-bit register + data byte to I2C slave
85 * @details Performs standard I2C write transaction with automatic STOP condition:
86 * 1. Wait for bus available (ISR.BUSY clear)
87 * 2. Configure CR2: 2 bytes, AUTOEND, START condition
88 * 3. Send register address (TXDR)
89 * 4. Send data byte (TXDR)
90 * 5. Wait for TXE (transmitter empty); AUTOEND triggers STOP
91 *
92 * ### Transaction Sequence
93 * ```
94 * START [slave_addr(W)] ACK [register_addr] ACK [data_byte] ACK STOP
95 * ```
96 * - START: Generated by CR2.START
97 * - slave_addr: Shifted left 1 bit by hardware; (slave<<1) or slave|0
98 * - AUTOEND: Automatically generates STOP after NBYTES bytes sent
99 *
100 * @param slave - 7-bit I2C slave address (e.g., 0x5D for MAX30101)
101 * Format: [D6:D0] (do NOT pre-shift)
102 * @param addr - Register address (0x00-0xFF) in slave device
103 * @param data - Single data byte to write (0x00-0xFF)
104 * @return void
105 * @retval N/A (blocking until complete or bus error)
106 *
107 * @timing
108 * - Bus arbitration: 1-2 µs
109 * - Byte transmission (2×): ~10 µs each = 20 µs
110 * - ACK/STOP wait: ~10 µs
111 * - **Total latency**: ~30-50 µs (typical)
112 * - Maximum timeout (bus error): ~1 sec (hardware dependent)
113 *
114 * @flags_monitored
115 * - BUSY: Wait until clear (bus available)
116 * - TXIS: Transmit interrupt status, fires per byte ready to send
117 * - TXE: Transmitter empty, fires after last byte enqueued
118 * - AUTOEND: Firmware-set flag; hardware auto-clears and generates STOP
119 *
120 * @error_conditions
121 * - **NAK received**: Bus hangs in BUSY state; requires I2C reset (see datasheet)
122 * - **Slave not present**: Timeout; TXE may never set
123 * - **SDA stuck low**: Requires manual GPIO toggle to recover
124 *
125 * @usage_example
126 * ```
127 * // Write LED power control register to MAX30101
128 * I2C1_Write(0x5D, LED1_PAMPLI, 0x3F); // slave=0x5D, addr=0x0C, data=0x3F
129 * ```
130 *
131 * @see I2C1_Read, I2C specification (NXP UM10204)
132 */
133void I2C1_Write(uint8_t slave, uint8_t addr, uint8_t data){
134 // Wait for bus to be available
135 while(I2C1->ISR & I2C_ISR_BUSY);
136 // Set up transfer: slave address, 2 bytes, AUTOEND, START
137 I2C1->CR2 = 0x00;
138 I2C1->CR2 = I2C_CR2_AUTOEND | (2<<16) | (slave<<1) | I2C_CR2_START;
139 // Send register address
140 while(!(I2C1->ISR & I2C_ISR_TXIS));
141 I2C1->TXDR = addr;
142 // Send data byte
143 while(!(I2C1->ISR & I2C_ISR_TXIS));
144 I2C1->TXDR = data;
145 // Wait for transmission complete
146 while(!(I2C1->ISR & I2C_ISR_TXE));
147}

149/**
150 * @brief Master read multiple bytes from I2C slave register (repeated START)
151 * @details Performs combined write-read transaction without bus release:
152 * **Phase 1 (Write)**:
153 * 1. Configure CR2: 1 byte, NO AUTOEND, START condition
154 * 2. Send register address
155 * 3. Wait for TC (transfer complete, not TXE)
156 * **Phase 2 (Repeated START + Read)**:
157 * 1. Configure CR2: RD_WRN=1, size bytes, AUTOEND, START condition
158 * 2. Read each byte from RXDR (data available via RXNE flag)
159 * 3. Wait for STOPF (AUTOEND triggers STOP)
160 *
161 * ### Transaction Sequence
162 * ```
163 * START [slave_addr(W)] ACK [register_addr] ACK
164 * RESTART [slave_addr(R)] ACK [data_0] ACK [data_1] ACK ... STOP
165 * ```
166 * - Repeated START: Avoids releasing bus; allows atomic address+read
167 * - RD_WRN: Set for read phase; hardware handles address bit automatically
168 *
169 * @param slave - 7-bit I2C slave address (e.g., 0x5D for MAX30101)
170 * Format: [D6:D0] (do NOT pre-shift)
171 * @param addr - Register address in slave device to read from (e.g., 0x07 for MAX30101 FIFO)
172 * @param data - [out] Pointer to buffer for received bytes
173 * Must be allocated by caller with capacity ≥ size bytes
174 * @param size - [in] Number of bytes to read from slave (1-255)
175 * Typical values: 1 (single register), 6 (MAX30101 FIFO sample)
176 * @return void
177 * @retval N/A (blocking until complete or error)
178 *
179 * @timing
180 * - Write address phase: ~15-20 µs
181 * - Repeated START latency: ~10-15 µs
182 * - Read phase (per byte): ~10-15 µs
183 * - **Total latency**: 100+ 30×size µs (typical)
184 * - Example: Read 6 bytes ≈ 180-200 µs
185 * - Dominates any sensor timing; I2C is bottleneck at 100 Hz sampling
186 *
187 * @flags_monitored
188 * - BUSY: Wait until clear before write phase
189 * - TXIS: Ready for register address byte
190 * - TC (not TXE!): Critical; signals address byte sent, allowing repeated START
191 * - RXNE: Data byte available in RXDR
192 * - STOPF: AUTOEND triggered; read phase complete
193 *
194 * @error_conditions
195 * - **Register address invalid**: Slave NAKs; BUSY remains set
196 * - **No slave at address**: BUSY hangs indefinitely (no timeout)
197 * - **Buffer overflow**: Caller must ensure data[] size ≥ size parameter
198 * - **Timeout on RXNE**: Slave not responding; function blocks forever
199 *
200 * @performance_note
201 * - Read 32 FIFO samples (192 bytes): ~6 milliseconds
202 * - Pure I2C time at 100 Hz ÷ 32 samples = dominant ISR latency
203 * - Consider DMA or firmware optimization for high-speed sampling
204 *
205 * @usage_example
206 * ```
207 * // Read 6 bytes from MAX30101 FIFO (one sample, 3 channels × 2 bytes)
208 * uint8_t fifo_data[6];
209 * I2C1_Read(0x5D, 0x07, fifo_data, 6);
210 * // fifo_data[0] = red[MSB], fifo_data[1] = red[LSB], ...
211 * ```
212 *
213 * @see I2C1_Write, I2C specification (repeated START section)
214 */
215void I2C1_Read(uint8_t slave, uint8_t addr, uint8_t *data, uint8_t size){
216 // Wait for bus to be available
217 while(I2C1->ISR & I2C_ISR_BUSY);
218 // Send register address (write 1 byte, no AUTOEND for repeated START)
219 I2C1->CR2 = 0x00;
220 I2C1->CR2 = (1<<16) | (slave<<1) | I2C_CR2_START;
221 // Send register address
222 while(!(I2C1->ISR & I2C_ISR_TXIS));
223 I2C1->TXDR = addr;
224 // Wait for transfer complete (TC flag)
225 while(!(I2C1->ISR & I2C_ISR_TC));
226 // Read data (multiple bytes, AUTOEND, RD_WRN=1)
227 I2C1->CR2 = 0x00;
228 I2C1->CR2 = I2C_CR2_AUTOEND | I2C_CR2_RD_WRN | (size<<16) | (slave<<1) | I2C_CR2_START;
229 // Read each byte
230 for(uint8_t i = 0; i < size; i++){
231 // Wait for data ready
232 while(!(I2C1->ISR & I2C_ISR_RXNE));
233 data[i] = I2C1->RXDR;
234 }
235 // Wait for stop condition
236 while(!(I2C1->ISR & I2C_ISR_STOPF));
237}
